    --agregar 
        export default defineConfig({
            plugins: [react()],
            resolve: {
                alias: [{ find: '@', replacement: path.resolve(__dirname, 'src') }],
                // cada vez que en un impor se vea un "@" lo toma como "src"
            },
        })
    --agregar, en tsconfig.json en compiler options
        "paths": {
            "@/*": ["./src/*"]
